689// Modified UIWindow_Render to support high-quality antialiased shapes
690int UIWindow_Render(UIWindow* window) {
691 if (!window || !window->sdlRenderer) return -1;
692 int rw, rh;
693 SDL_GetWindowSize(window->sdlWindow, &rw, &rh);
694
695 // Initialize TTF once
696 static int ttfInited = 0;
697 if (!ttfInited) {
698 if (TTF_Init() != 1)
699 fprintf(stderr, "TTF_Init error: %s\n", SDL_GetError());
700 ttfInited = 1;
701 }
702
703 // FPS calculation
704 static Uint64 lastCounter = 0, frameCount = 0, freq = 0;
705 if (!freq) freq = SDL_GetPerformanceFrequency();
706 frameCount++;
707 Uint64 cur = SDL_GetPerformanceCounter();
708 if (!lastCounter) lastCounter = cur;
709 if (cur - lastCounter >= freq) {
710 window->framerate = (float)frameCount / ((cur - lastCounter) / (float)freq);
711 lastCounter = cur;
712 frameCount = 0;
713 if (window->events) {
714 UIEventData ev = {0};
715 ev.framerate.fps = window->framerate;
716 ev.children = window->children;
717 ev.type = UI_EVENT_FRAMERATE_CHANGED;
718 UIWindow_EmitEvent(window, UI_EVENT_FRAMERATE_CHANGED, ev);
719 }
720 }
721
722 // Prepare or resize smooth texture with higher resolution for antialiasing
723 int upscale = 4; // Increased upscale factor for even better quality
724 int tw = rw * upscale, th = rh * upscale;
725 if (!g_smoothTexture || g_smoothW != tw || g_smoothH != th) {
726 if (g_smoothTexture) SDL_DestroyTexture(g_smoothTexture);
727 g_smoothTexture = SDL_CreateTexture(window->sdlRenderer,
728 SDL_PIXELFORMAT_RGBA8888, SDL_TEXTUREACCESS_TARGET, tw, th);
729 if (!g_smoothTexture) {
730 fprintf(stderr, "Failed to create smooth texture: %s\n", SDL_GetError());
731 return -1;
732 }
733 g_smoothW = tw;
734 g_smoothH = th;
735 SDL_SetTextureScaleMode(g_smoothTexture, SDL_SCALEMODE_LINEAR);
736 }
737
738 SDL_SetRenderTarget(window->sdlRenderer, g_smoothTexture);
739 SDL_SetRenderScale(window->sdlRenderer, (float)upscale, (float)upscale);
740 SDL_SetRenderDrawBlendMode(window->sdlRenderer, SDL_BLENDMODE_BLEND);
741 SDL_SetRenderDrawColor(window->sdlRenderer,
742 (Uint8)window->backgroundColor.r,
743 (Uint8)window->backgroundColor.g,
744 (Uint8)window->backgroundColor.b,
745 (Uint8)SDL_clamp((int)(window->backgroundColor.a * 255), 0, 255));
746 SDL_RenderClear(window->sdlRenderer);
747
748 // Render all UI elements with high-quality antialiasing
749 if (window->children) {
750 UIChildren_SortByZ(window->children);
751 for (int i = 0; i < window->children->count; i++) {
752 UIWidget *el = window->children->children[i];
753 if (!el || !el->visible || !el->data) continue;
754 if (el->alignment) UIAlignment_Align(el);
755 UIWidgetBase *base = (UIWidgetBase*)el->data;
756
757 if (!strcmp(base->__widget_type, UI_WIDGET_RECTANGLE)) {
758 UIRectangle *rect = (UIRectangle*)base;
759 SDL_FRect rf = {el->x, el->y, *el->width, *el->height};
760 ApplyMargins(&rf, rect->marginLeft, rect->marginTop,
761 rect->marginRight, rect->marginBottom);
762
763 // Handle special case for circles (detect when width == height and radius is half width)
764 if (rf.w > 0 && rf.h > 0) {
765 if (fabsf(rf.w - rf.h) < 0.5f && fabsf(rect->radius - rf.w/2) < 0.5f) {
766 // This is effectively a circle - use our specialized circle renderer
767 float centerX = rf.x + rf.w/2;
768 float centerY = rf.y + rf.h/2;
769 float radius = rf.w/2;
770
771 // Draw border if needed
772 if (rect->borderWidth > 0) {
773 DrawCircleTextured(window->sdlRenderer, centerX, centerY,
774 radius, rect->borderColor);
775 radius -= rect->borderWidth;
776 }
777
778 // Draw inner circle
779 if (radius > 0) {
780 DrawCircleTextured(window->sdlRenderer, centerX, centerY,
781 radius, rect->color);
782 }
783 } else {
784 // Regular rounded rectangle - use existing function but with improved quality
785 SDL_SetRenderDrawBlendMode(window->sdlRenderer, SDL_BLENDMODE_BLEND);
786 DrawRoundedRectWithAlphaAA(window->sdlRenderer, rf,
787 rect->color, rect->radius,
788 (int)rect->borderWidth, rect->borderColor);
789 }
790 }
791 }
792 else if (!strcmp(base->__widget_type, UI_WIDGET_TEXT)) {
793 UIText *twid = (UIText*)base;
794 if (!twid->text || !twid->fontFamily || !*twid->text || !twid->textLength || !twid->fontSize)
795 continue;
796
797 if (!twid->__SDL_textTexture) {
798 TTF_Font *font = GetFont(twid->fontFamily, twid->fontSize * 1.5f); // Use higher res font
799 if (font) {
800 // Enable best quality rendering
801 TTF_SetFontHinting(font, TTF_HINTING_LIGHT);
802
803 SDL_Color sc = {(Uint8)twid->color.r,
804 (Uint8)twid->color.g,
805 (Uint8)twid->color.b,
806 (Uint8)SDL_clamp((int)(twid->color.a * 255), 0, 255)};
807
808 // Use LCD subpixel rendering for maximum text quality if available
809 // Or fallback to blended mode
810 SDL_Surface *surf = TTF_RenderText_Blended(
811 font, twid->text, twid->textLength, sc);
812
813 if (surf) {
814 twid->__SDL_textTexture = SDL_CreateTextureFromSurface(
815 window->sdlRenderer, surf);
816 SDL_DestroySurface(surf);
817 }
818 }
819 }
820
821 if (!twid->__SDL_textTexture) continue;
822
823 // Set the best quality scaling mode for text
824 SDL_SetTextureScaleMode(twid->__SDL_textTexture, SDL_SCALEMODE_LINEAR);
825
826 // Query texture size
827 float txw, txh;
828 SDL_GetTextureSize(twid->__SDL_textTexture, &txw, &txh);
829
830 // Apply padding
831 float pw = txw + twid->paddingLeft + twid->paddingRight;
832 float ph = txh + twid->paddingTop + twid->paddingBottom;
833 SDL_FRect bg = {el->x, el->y, pw, ph};
834
835 // Draw background with high quality
836 if (fabsf(twid->background->radius - bg.w/2) < 0.5f &&
837 fabsf(bg.w - bg.h) < 0.5f) {
838 // Background is a circle
839 float centerX = bg.x + bg.w/2;
840 float centerY = bg.y + bg.h/2;
841 DrawCircleTextured(window->sdlRenderer, centerX, centerY,
842 bg.w/2, twid->background->color);
843 } else {
844 // Background is a rounded rectangle
845 DrawRoundedRectWithAlphaAA(window->sdlRenderer, bg,
846 twid->background->color,
847 twid->background->radius,
848 (int)twid->background->borderWidth,
849 twid->background->borderColor);
850 }
851
852 // Draw text inside padded area with subpixel precision
853 SDL_FRect dst = {bg.x + twid->paddingLeft,
854 bg.y + twid->paddingTop,
855 txw, txh};
856 SDL_RenderTexture(window->sdlRenderer, twid->__SDL_textTexture,
857 NULL, &dst);
858 }
859 // Add support for custom UI widget types here
860 }
861 }
862
863 // Switch to the default render target and draw the upscaled texture with best quality scaling
864 SDL_SetRenderTarget(window->sdlRenderer, NULL);
865 SDL_SetRenderScale(window->sdlRenderer, 1.0f, 1.0f);
866
867 // Use high-quality downscaling
868 SDL_SetTextureScaleMode(g_smoothTexture, SDL_SCALEMODE_LINEAR);
869
870 SDL_FRect screenRect = {0, 0, (float)rw, (float)rh};
871 SDL_RenderTexture(window->sdlRenderer, g_smoothTexture, NULL, &screenRect);
872 SDL_RenderPresent(window->sdlRenderer);
873
874 return 0;
875}

1016void UIWindow_SetProperty(UIWindow* window, const char* property, void* value) {
1017 if (!window || !property || !value) return;
1018
1019 // Verify if the property already exists
1020 for (unsigned int i = 0; i < window->__ui_props.count; i++) {
1021 if (window->__ui_props.props[i] && strcmp(window->__ui_props.props[i]->key, property) == 0) {
1022 window->__ui_props.props[i]->value = value;
1023 return;
1024 }
1025 }
1026
1027 // Verify if we need to expand the properties array
1028 if (window->__ui_props.count >= window->__ui_props.capacity) {
1029 unsigned int new_capacity = window->__ui_props.capacity * 2;
1030 UIProp** new_props = (UIProp**)realloc(window->__ui_props.props, new_capacity * sizeof(UIProp*));
1031 if (!new_props) {
1032 fprintf(stderr, "Failed to reallocate memory for UI properties\n");
1033 return;
1034 }
1035 window->__ui_props.props = new_props;
1036 window->__ui_props.capacity = new_capacity;
1037
1038 // Initialize new properties to NULL
1039 for (unsigned int i = window->__ui_props.count; i < new_capacity; i++) {
1040 window->__ui_props.props[i] = NULL;
1041 }
1042 }
1043
1044 // Add the new property
1045 UIProp* new_prop = (UIProp*)malloc(sizeof(UIProp));
1046 if (!new_prop) {
1047 fprintf(stderr, "Failed to allocate memory for new UIProp\n");
1048 return;
1049 }
1050
1051 char* key_copy = _strdup(property);
1052 if (!key_copy) {
1053 fprintf(stderr, "Failed to allocate memory for property key\n");
1054 free(new_prop);
1055 return;
1056 }
1057 new_prop->key = key_copy;
1058
1059 new_prop->value = value;
1060 window->__ui_props.props[window->__ui_props.count++] = new_prop;
1061}
