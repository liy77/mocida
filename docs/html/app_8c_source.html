12void HandleEvent(UIApp* app, SDL_Event* event) {
13 if (!app || !app->window) return;
14
15 switch (event->type) {
16 case SDL_EVENT_DROP_POSITION: {
17 float x = event->drop.x;
18 float y = event->drop.y;
19
20 app->window->x = x;
21 app->window->y = y;
22 break;
23 }
24 case SDL_EVENT_WINDOW_RESIZED: {
25 int new_width = event->window.data1;
26 int new_height = event->window.data2;
27
28 app->window->width = new_width;
29 app->window->height = new_height;
30
31 // Update the main widget size for layout purposes
32 if (app->mainWidget == NULL) {
33 fprintf(stderr, "Main widget is NULL\n");
34 } else {
35 UIWidget_SetSize(app->mainWidget, (float)new_width, (float)new_height);
36 }
37
38 if (app->window->sdlRenderer) {
39 if (SDL_SetRenderLogicalPresentation(app->window->sdlRenderer, new_width, new_height, SDL_LOGICAL_PRESENTATION_LETTERBOX) != 1) {
40 fprintf(stderr, "SDL_SetRenderLogicalPresentation Error: %s\n", SDL_GetError());
41 }
42 } else {
43 fprintf(stderr, "SDL_Renderer is NULL\n");
44 }
45
46 break;
47 }
48 default:
49 break;
50 }
51}

59UIApp* UIApp_Create(const char* title, int width, int height) {
60 UIApp* app = (UIApp*)malloc(sizeof(UIApp));
61 if (!app) {
62 fprintf(stderr, "Failed to allocate memory for UIApp\n");
63 return NULL;
64 }
65
66 // Initialize properties before creating the window, to avoid accessing uninitialized memory
67 app->mainWidget = NULL;
68 app->window = NULL;
69 app->backgroundColor = UI_COLOR_WHITE;
70
71 UIWidget* mainWidget = widgc(NULL);
72 if (!mainWidget) {
73 fprintf(stderr, "Failed to create main widget\n");
74 free(app);
75 return NULL;
76 }
77 UIWidget_SetSize(mainWidget, (float)width, (float)height);
78 app->mainWidget = mainWidget;
79
80 app->window = UIWindow_Create(title, width, height);
81 if (app->window == NULL) {
82 fprintf(stderr, "UIWindow_Create returned NULL\n");
83 UIWidget_Destroy(mainWidget);
84 free(app);
85 return NULL;
86 }
87
88 UIApp_SetBackgroundColor(app, UI_COLOR_WHITE);
89 return app;

160void UIApp_SetRenderDriver(UIApp* app, UIRenderDriver renderDriver) {
161 if (!app || !app->window || !app->window->sdlWindow) return;
162
163 // Store the current renderer to free it only if we successfully create a new one
164 SDL_Renderer* currentRenderer = app->window->sdlRenderer;
165 SDL_Renderer* newRenderer = NULL;
166 const char* driverName = NULL;
167
168 switch (renderDriver) {
169 case UI_RENDER_OPENGL:
170 SDL_SetHint(SDL_HINT_RENDER_DRIVER, "opengl");
171 driverName = "opengl";
172 break;
173 case UI_RENDER_SOFTWARE:
174 SDL_SetHint(SDL_HINT_RENDER_DRIVER, "software");
175 driverName = "software";
176 break;
177 case UI_RENDER_VULKAN:
178 SDL_SetHint(SDL_HINT_RENDER_DRIVER, "vulkan");
179 driverName = "vulkan";
180 break;
181 case UI_RENDER_3D9:
182 // Used for Direct3D 9 - For legacy systems
183 SDL_SetHint(SDL_HINT_RENDER_DRIVER, "direct3d");
184 driverName = "direct3d";
185 break;
186 case UI_RENDER_3D11:
187 SDL_SetHint(SDL_HINT_RENDER_DRIVER, "direct3d11");
188 driverName = "direct3d11";
189 break;
190 case UI_RENDER_3D12:
191 SDL_SetHint(SDL_HINT_RENDER_DRIVER, "direct3d12");
192 driverName = "direct3d12";
193 break;
194 case UI_RENDER_GPU:
195 SDL_SetHint(SDL_HINT_RENDER_DRIVER, "gpu");
196 driverName = "gpu";
197 break;
198 #ifdef __APPLE__
199 case UI_RENDER_METAL:
200 // Note: Metal is only available on macOS and iOS
201 SDL_SetHint(SDL_HINT_RENDER_DRIVER, "metal");
202 driverName = "metal";
203 break;
204 #endif
205 default:
206 fprintf(stderr, "Unknown render driver\n");
207 return;
208 }
209
210 // Create a new renderer with the specified driver
211 if (driverName) {
212 newRenderer = SDL_CreateRenderer(app->window->sdlWindow, driverName);
213 if (newRenderer) {
214 // Destroys the current renderer only if the new renderer is created successfully
215 if (currentRenderer) {
216 SDL_DestroyRenderer(currentRenderer);
217 }
218 app->window->sdlRenderer = newRenderer;
219 } else {
220 fprintf(stderr, "Failed to create renderer with driver %s: %s\n",
221 driverName, SDL_GetError());
222 }
223 }

267void UIApp_Run(UIApp* app) {
268 if (!app || !app->window) return;
269 SDL_Event e;
270 int frameDelayMs = 14; // Approximating 71 FPS (1000ms/71 14ms)
271
272 while (app->window->visible) {
273 while (SDL_PollEvent(&e)) {
274 if (e.type == SDL_EVENT_QUIT) {
275 app->window->visible = 0;
276 }
277 HandleEvent(app, &e);
278 }
279
280 // Render and wait
281 UIWindow_Render(app->window);
282
283 SDL_Delay(frameDelayMs);
284 }
